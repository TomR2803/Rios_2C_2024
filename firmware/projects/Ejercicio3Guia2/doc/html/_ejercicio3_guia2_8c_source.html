190static void MostrarTask(void *pvParameter){
191 while(true){
192 //Checkeo si la tarea de medición está activa
193 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
194 printf("mostrarTask\n");
195 if(global_on == true)
196 {
197 // Enciendo LEDs según la distancia
198 if(distancia < 10){
199 LedsOffAll();
200 }
201 else if (distancia >= 10 && distancia < 20){
202 LedOn(LED_1);
203 LedOff(LED_2);
204 LedOff(LED_3);
205 }
206 else if(distancia >= 20 && distancia < 30){
207 LedOn(LED_1);
208 LedOn(LED_2);
209 LedOff(LED_3);
210 }
211 else if(distancia >= 30){
212 LedOn(LED_1);
213 LedOn(LED_2);
214 LedOn(LED_3);
215 }
216
217 // Si el modo hold no está activado, muestro distancia en la pantalla LCD
218 if(global_hold == false){
219 LcdItsE0803Write(distancia);
220 }
221 }
222 else
223 {
224 // Apagar LEDs y pantalla LCD si on está apagado
225 LedOff(LED_1);
226 LedOff(LED_2);
227 LedOff(LED_3);
228 LcdItsE0803Off();
229 }
230 }
