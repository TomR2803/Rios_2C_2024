154static void MostrarTask(void *pvParameter){
155 while(true){
156 //Checkeo si la tarea de medición está activa
157 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
158 printf("mostrarTask\n");
159 if(global_on == true)
160 {
161 // Enciendo LEDs según la distancia
162 if(distancia < 10){
163 LedsOffAll();
164 }
165 else if (distancia >= 10 && distancia < 20){
166 LedOn(LED_1);
167 LedOff(LED_2);
168 LedOff(LED_3);
169 }
170 else if(distancia >= 20 && distancia < 30){
171 LedOn(LED_1);
172 LedOn(LED_2);
173 LedOff(LED_3);
174 }
175 else if(distancia >= 30){
176 LedOn(LED_1);
177 LedOn(LED_2);
178 LedOn(LED_3);
179 }
180
181 // Si el modo hold no está activado, muestro distancia en la pantalla LCD
182 if(global_hold == false){
183 LcdItsE0803Write(distancia);
184 }
185 }
186 else
187 {
188 // Apagar LEDs y pantalla LCD si on está apagado
189 LedOff(LED_1);
190 LedOff(LED_2);
191 LedOff(LED_3);
192 LcdItsE0803Off();
193 }
194 }
