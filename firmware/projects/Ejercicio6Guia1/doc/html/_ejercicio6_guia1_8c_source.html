77void escribirBCDEnGPIO(uint8_t bcd, gpioConf_t *pines) {
78
79// Itera sobre el rango de pines GPIO desde MAX_PIN-1 hasta 0
80 for(uint8_t i=0;i<MAX_PIN;i++)
81 {
82 // Verifica si el bit menos significativo de bcd es 1
83 if(bcd%2)
84 // Si es 1, activa el pin GPIO correspondiente
85 GPIOOn(pines[MAX_PIN-1-i].pin);
86 else
87 // Si es 0, desactiva el pin GPIO correspondiente
