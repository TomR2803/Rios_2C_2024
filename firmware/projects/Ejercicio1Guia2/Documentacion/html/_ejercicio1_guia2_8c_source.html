164static void MostrarTask(void *pvParameter){
165 printf("Mostrar\n");
166 while(true){
167 //Checkeo si la tarea de medición está activa
168 if(global_on == true)
169 {
170 // Enciendo LEDs según la distancia
171 if(distancia < 10){
172 LedsOffAll();
173 }
174 else if (distancia >= 10 && distancia < 20){
175 LedOn(LED_1);
176 LedOff(LED_2);
177 LedOff(LED_3);
178 }
179 else if(distancia >= 20 && distancia < 30){
180 LedOn(LED_1);
181 LedOn(LED_2);
182 LedOff(LED_3);
183 }
184 else if(distancia >= 30){
185 LedOn(LED_1);
186 LedOn(LED_2);
187 LedOn(LED_3);
188 }
189
190 // Si el modo hold no está activado, muestro distancia en la pantalla LCD
191 if(global_hold == false){
192 LcdItsE0803Write(distancia);
193 }
194 }
195 else
196 {
197 // Apagar LEDs y pantalla LCD si on está apagado
198 LedOff(LED_1);
199 LedOff(LED_2);
200 LedOff(LED_3);
201 LcdItsE0803Off();
202 }
203 vTaskDelay(TASK_DELAY / portTICK_PERIOD_MS);
204 }
205}
