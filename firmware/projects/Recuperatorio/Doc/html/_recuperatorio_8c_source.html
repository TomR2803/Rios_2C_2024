215static void MedirTask(void *pvParameter){
216
217 while (true)
218 { printf("medirTask\n");
219 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
220 //mido la distancia con el sensor y le asigno el valor
221 //en cm a la variable distancia
222 DistanciaAnterior=distancia;
223 distancia = HcSr04ReadDistanceInCentimeters();
224
225 //verifico que la distancia sea menor o igual a 10 metros
226 //para empezar a medir la velocidad
227 if(distancia<=1000){
228
229 velocidad=((distancia-DistanciaAnterior)/100)/0.1;//calculo de la velocidad usando el periodo de muestreo
230 //como variacion de tiempo, para que la velocidad sea en m/s
231 //y paso la medicion del sensor a metros dividiendo por 100
232
233 //si la velocidad medida es mayor a la maxima registrada
234 //se cambia la velocidad maxima registrada
235 if(velocidad>VelocidadMaxima){
236 VelocidadMaxima=velocidad;
237 }
238
239 if(velocidad=0){
240 //si la velocidad del vehiculo es 0, es decir, si este se detiene
241 //cambio pesar a true, para empezar a pesar el vehiculo
242 pesar=true;
243 }
244 else{
245 pesar=false;
246 }
247
248 }
249 //si la distancia medida es mayor a 10 metros
250 //cambio pesar a false, para detener la pesa del vehiculo
251 //y cambio velocidad y VelocidadMaxima a 0

253 pesar=false;
254 velocidad=0;
255 VelocidadMaxima=0;
